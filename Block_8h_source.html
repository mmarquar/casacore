43 // <summary>simple 1-D array</summary>
44 // <use visibility=export>
45 //
46 // <reviewed reviewer="UNKNOWN" date="before2004/08/25" tests="" demos="">
47 // </reviewed>
48 //
49 // <etymology>
50 // This should be viewed as a <em>block</em> of memory without sophisticated
51 // manipulation functions. Thus it is called <src>Block</src>.
52 // </etymology>
53 //
54 // <synopsis>
55 // <src>Block<T></src> is a simple templated 1-D array class. Indices are always
56 // 0-based. For efficiency reasons, no index checking is done unless the
57 // preprocessor symbol <src>AIPS_ARRAY_INDEX_CHECK</src> is defined.
58 // <src>Block<T></src>'s may be assigned to and constructed from other
59 // <src>Block<T></src>'s.
60 // As no reference counting is done this can be an expensive operation, however.
61 //
62 // The net effect of this class is meant to be unsurprising to users who think
63 // of arrays as first class objects. The name "Block" is intended to convey
64 // the concept of a solid "chunk" of things without any intervening "fancy"
65 // memory management, etc. This class was written to be
66 // used in the implementations of more functional Vector, Matrix, etc. classes,
67 // although it is expected <src>Block<T></src> will be useful on its own.
68 //
69 // The Block class should be efficient. You should normally use <src>Block</src>.
70 //
71 // <note role=warning> If you use the assignment operator on an element of this
72 // class, you may leave dangling references to pointers released from
73 // <src>storage()</src>.
74 // Resizing the array will also have this effect if the underlying storage
75 // is actually affected.
76 // </note>
77 //
78 // If index checking is turned on, an out-of-bounds index will
79 // generate an <src>indexError<uInt></src> exception.
80 // </synopsis>
81 //
82 // <example>
83 // <srcblock>
84 // Block<Int> a(100,0); // 100 ints initialized to 0
85 // Block<Int> b; // 0-length Block
86 // // ...
87 // b = a; // resize b and copy a into it
88 // for (size_t i=0; i < a.nelements(); i++) {
89 // a[i] = i; // Generate a sequence
90 // // with Vectors, could simply say "indgen(myVector);"
91 // }
92 // b.set(-1); // All positions in b have the value -1
93 // b.resize(b.nelements()*2); // Make b twice as long, by default the old
94 // // elements are copied over, although this can
95 // // be defeated.
96 // some_c_function(b.storage()); // Use a fn that takes an
97 // // <src>Int *</src> pointer
98 // </srcblock>
99 // </example>
100 //

104  // Set the trace size. The (de)allocation of Blocks with >= sz elements

106  // A value 0 means no tracing.
107  static void setTraceSize (size_t sz);
108 protected:
109  // Write alloc and free trace messages.
110  static void doTraceAlloc (const void* addr, size_t nelem,
111  DataType type, size_t sz);
112  static void doTraceFree (const void* addr, size_t nelem,
113  DataType type, size_t sz);
114 protected:
115  static size_t itsTraceSize;
116 };

195  // <src>takeOverStorage</src> is <src>True</src>, The Block assumes that
196  // it owns the pointer, i.e. that it is safe to <src>delet[]</src> it when

227  // Resizes the Block. If <src>n == nelements()</src> resize just returns. If
228  // a larger size is requested (<src>n > nelements()</src>) the Block always
229  // resizes. If the requested size is smaller (<src>n < nelements()</src>),
230  // by default the Block does not resize smaller, although it can be
231  // forced to with <src>forceSmaller</src>. The reasoning behind this is that
232  // often the user will just want a buffer of at least a certain size,
233  // and won't want to pay the cost of multiple resizings.
234  // <srcblock>
235  // Block<float> bf(100, 0.0);
236  // bf.resize(10); // bf.nelements() == 100
237  // bf.resize(10, True) // bf.nelements() == 10
238  // bf.resize(200) // bf.nelements() == 200
239  // </srcblock>
240  // Normally the old elements are copied over (although if the
241  // Block is lengthened the trailing elements will have undefined
242  // values), however this can be turned off by setting copyElements to
243  // False.
244  //
245  // This is written as three functions because default parameters do
246  // not always work properly with templates.
247  // <group>
248  void resize(size_t n, Bool forceSmaller=False, Bool copyElements=True) {
249  if (!(n == npts || (n < npts && forceSmaller == False))) {
250  T *tp = n > 0 ? new T[n] : 0;
251  traceAlloc (tp, n);
252  if (copyElements) {
253  size_t nmin = npts < n ? npts : n; // Don't copy too much!
254  objcopy(tp, array, nmin);
255  };
256  if (array && destroyPointer) { // delete...
257  traceFree (array, npts);
258  delete [] array;
259  array = 0;
260  };
261  npts = n;
262  destroyPointer = True;
263  array = tp; // ... and swap pointer
264  };
265  }
266  // </group>

268  // Remove a single element from the Block. If forceSmaller is True this
269  // will resize the Block and hence involve new memory allocations. This is
270  // relatively expensive so setting forceSmaller to False is preferred. When
271  // forcesmaller is False the Block is not resized but the elements with an
272  // index above the removed element are shuffled down by one. For backward
273  // compatibility forceSmaller is True by default.
274  // <group>
275  void remove(size_t whichOne, Bool forceSmaller=True) {
276  if (whichOne >= npts) {
277 #if defined(AIPS_ARRAY_INDEX_CHECK)
278  throw(indexError<uInt>(whichOne, "Block::remove() - "
279  "index out of range"));
280 #else
281  return;
282 #endif
283  }
284  if (forceSmaller == True) {
285  T *tp = new T[npts - 1];
286  traceAlloc (array, npts-1);
287  objcopy(tp, array, whichOne);
288  objcopy(tp+whichOne, array + whichOne + 1, npts - whichOne - 1);
289  if (array && destroyPointer) {
290  traceFree (array, npts);
291  delete [] array;
292  array = 0;
293  };
294  npts--;
295  array = tp;
296  destroyPointer = True;
297  } else objmove(array+whichOne, array + whichOne + 1, npts - whichOne - 1);
298  }
299  // </group>

301  // Replace the internal storage with a C-array (i.e. pointer).
302  // If <src>takeOverStorage</src> is True, The Block assumes that it
303  // owns the pointer, i.e. that it is safe to <src>delete[]</src> it when the
304  // <src>Block</src>is destructed, otherwise the actual storage is not destroyed.
305  // If true, storagePointer is set to <src>NULL</src>.
306  void replaceStorage(size_t n, T *&storagePointer, Bool takeOverStorage=True) {
307  if (array && destroyPointer) {
308  traceFree (array, npts);
309  delete [] array;
310  array = 0;
311  };
312  npts = n;
313  array = storagePointer;
314  destroyPointer = takeOverStorage;
315  if (destroyPointer) storagePointer = 0;
316  };
