45 // <summary>
46 // Quantities (i.e. dimensioned values)
47 // </summary>
48 
49 // <use visibility=export>
50 
51 // <reviewed reviewer="UNKNOWN" date="before2004/08/25" tests="tQuantum">
52 // </reviewed>
53 //
54 // <prerequisite>
55 // <li> <linkto class=Unit>Unit</linkto>
56 // </prerequisite>
57 //
58 // <etymology>
59 // A Quantity is defined as a single Double value with attached units.
60 // From this definition the templated Quantum class arose, to have non-Double,
61 // non-scalar quantities.
62 // </etymology>
63 //
64 // <synopsis>
65 // Quantities are values with a unit. Their basic specification can be one of
66 // two forms:
67 // <srcblock>
68 // Quantity( Double value, String unit); // or: Unit unit
69 // Quantum<Type> ( Type value, String unit) // or: Unit unit
70 // </srcblock>
71 //
72 // A unit is a string of known unit fields separated
73 // by 'space' or '.' (to indicate multiply) or '/' (to indicate divide).
74 // See the <linkto class=Unit>Unit</linkto> class for details.
75 //
76 // Example: km/s/(Mpc.s)2 is identical to km.s-1.Mpc-2.s-2
77 //
78 // <h3> Defining a Quantum </h3>
79 // The following list of constructors is available.
80 // <note role=tip>
81 // In the following 'String' can be replaced by 'Unit' everywhere. The
82 // only difference being a check for a legitimate unit string being executed
83 // if Unit specified (with exception if error)
84 // </note>
85 // <note role=tip>
86 // <src>'Quantum<Type>'</src> can, if Type equals Double, be replaced
87 // with 'Quantity'
88 //
89 // 'Type' can be any simple or non-simple arithmetic type.
90 //
91 // E.g. <src><Double>, <Complex>, <Vector<Double> ></src>
92 // </note>
93 // <ul>
94 // <li> <src>Quantum<Type>() value 0 generated</src>
95 // <li> <src>Quantum<Type>( Quantum<Type>) copy constructor</src>
96 // <li> <src>Quantum<Type>( Type factor) value factor generated</src>
97 // <li> <src>Quantum<Type>( Type factor, Unit unit) specified quantity</src>
98 // <li> <src>Quantum<Type>( Type factor, Quantum<any> quant) specified factor,</src>
99 // the unit from the quant
100 // </ul>
101 //
102 //
103 // <h3> Manipulating quantities </h3>
104 // <linkto group="QMath.h#Quantum mathematical operations">Mathematical operators and functions</linkto> and
105 // <linkto group="QLogical.h#Quantum logical operations">logical operations</linkto> (comparisons)
106 // are defined on Quantums. They are,
107 // of course, only available if the template Type supports them.
108 // <ul>
109 // <li> <src>= assignment of identical <type></src>
110 // <li> <src>* *= multiple two Quantums of same <type>, or Quantum and type</src>
111 // <li> <src>/ /= divide two Quantums of same <type>, or Quantum and type</src>
112 // note:
113 // In multiplication and division, and if <src><type></src> is scalar, the left or
114 // right-hand side can be of type <src><type></src> (e.g 2.*Quantity is allowed)
115 // <li> <src>+ += add two Quantums of same <type> or Quantum and type</src>
116 // and same unit dimensions (else exception)
117 // <li> - -= subtract (same as +)
118 // <li> - negate Quantum
119 // <li> + unary + on Quantum
120 // <li> <src>== != compare unit dimensions and value of same <type>. They will</src>
121 // be unequal if the units do not match or the values (possibly
122 // converted to common base units). All comparisons work also
123 // on a <src>Quantum<type> and <type></src>
124 // <li> <src>< > compare unit dimensions. Exception if no match,</src>
125 // else compare the values
126 // <li> <src><= >= ibid</src>
127 // <li> pow(Int) raise to an (integer) power
128 // </ul>
129 //
130 //
131 // <h3> Manipulating the value and/or units of quanta </h3>
132 // Quantities can be converted to other units by the following set of member
133 // functions:
134 // <ul>
135 // <li> convert() will convert the quantum to canonical units.
136 // E.g. given myval=Quantity(5.,"Jy"),
137 // myval.convert() will convert the qunatum to
138 // Quantity(5.e-26,"kg.s-2")
139 // <li> convert(Unit unit) will convert the quantum to the
140 // specified unit with any remaining dimensions
141 // expressed in canonical units. E.g given
142 // myval as above, myval.convert("W/cm") will
143 // make it Quantity(5.e-28,"W/cm.m-1.s")
144 // <li> <src>convert(Quantum<Type> quant) will convert the quantum</src>
145 // to the units of the specified quant with the
146 // same conversion rules as the previous one
147 // </ul>
148 // <note role=tip> All converting type methods (i.e. convert(), get() and
149 // getValue() with specified units), will automatically convert also from
150 // time to angle units (or v.v) if necessary, as long as they are simple. I.e.
151 // deg will be converted to h, but asking to convert m/s to m/deg will
152 // produce the standard conversion to m/deg.rad/s. </note>
153 //
154 // Quanta can be checked for having the correct unit dimensions (e.g. before
155 // addition or comparing) by the following two member functions, which will
156 // return a Bool value:
157 // <ul>
158 // <li> isConform(Unit unit)
159 // <li> <src>isConform(Quantum<Type> quant)</src>
160 // <li> check(UnitVal kind)
161 // </ul>
162 // or by an assertion, which will throw an exception:<br>
163 // <ul>
164 // <li> assure(UnitVal kind)
165 // </ul>
166 //
167 // The quantum can be retrieved with a change in units by:
168 // <ul>
169 // <li> get() will return the quantum converted to canonical units.
170 // E.g. given myval=Quantity(5.,"Jy"),
171 // myval.get() will return
172 // Quantity(5.e-26,"kg.s-2")
173 // <li> get(Unit unit) will return the quantum converted to the
174 // specified unit with any remaining dimensions
175 // expressed in canonical units. E.g given
176 // myval as above, myval.get("W/cm") will
177 // return it as Quantity(5.e-28,"W/cm.m-1.s")
178 // <li> <src>get(Quantum<Type> quant) will return the quantum converted</src>
179 // to the units of the specified quant with the
180 // same conversion rules as the previous one
181 // </ul>
182 //
183 // The value and units of a quantum can be set or retrieved separately by the
184 // following member functions:
185 // <ul>
186 // <li> getValue() return the value (as Type) of the quantum.
187 // <note role=tip> myval.get().getValue() will return the
188 // value of myval expressed in canonical units
189 // </note>
190 // <li> getValue(Unit unit) return the value (as converted to unit)
191 // <li> getUnit() return the String part of the unit of the
192 // quantum (use getFullUnit if interested in
193 // the complete Unit, e.g. for re-use)
194 // <li> getFullUnit() return the complete unit of the Quantum (use
195 // getUnit() if interested in String part only)
196 // <li> setValue(Type val) replace the value of the quantum with val,
197 // leaving the units the same
198 // <li> scale(Type val) multiply the value (leaving units same) by the
199 // specified value
200 // <li> setUnit(Unit unit) replace the units of the quantum, leaving
201 // the value the same.
202 // <li> <src>setUnit(Quantum<Type> quant) ibid</src>
203 // <li> set(String quantity) replace the value and unit as deduced from quantity
204 // </ul>
205 //
206 // The output operator (<src><<</src>) will produce the value of the quantum and its
207 // units. Given <src>Quantity myval(5.,"mJy"), << myval</src> will produce:
208 // <src>5.0 mJy</src>; while <src><< myval.get("yW/m2")</src>
209 // will produce: <src>.00005 yW/m2.s</src>.<br>
210 // The input operator (<src>>></src>, or the static read functions) will
211 // convert a String to a Quantum (quantity only for now). The analysis
212 // will do the following:
213 // <ul>
214 // <li> Check if it can be converted as a time/angle, if so use
215 // (<linkto class=MVAngle>MVAngle</linkto>)
216 // <li> Check if it can be used as a date/time. if so use
217 // (<linkto class=MVTime>MVTime</linkto>)
218 // <li> Interpret as a value with units
219 // </ul>
220 // <note role=caution> Since e.g. <em>12d</em> could be interpreted as
221 // being both an angle (12 degrees) or a quantity (12 days), the only way
222 // is to differentiate them with a decimal point (12.d will be days)</note>
223 //
224 // </synopsis>
225 //
226 // <example>
227 // An experiment has measured the energy of a photon in keV. The following will
228 // output the wavelength and frequency of this photon (see the
229 // <linkto class=QC">QC</linkto> class for quantity constants):
230 // <srcblock>
231 // #include <casacore/casa/Quanta.h>
232 // Double myval; // keV photon energy
233 // Quantity quant(myval,"keV"); // make quantity
234 // cout << "A photon with energy " << quant << endl
235 // << " has a frequency of "
236 // << (quant/QC::h)->get("GHz") << endl // h=Planck
237 // << " and a wavelength of "
238 // << (QC::c/quant/QC::h)->get("nm") // c=light velocity
239 // << " or " << QC::c/quant/QC::h << endl;
240 // </srcblock>
241 // </example>
242 //
243 // <motivation>
244 // Major use is foreseen in all calculations with observed data.
245 // </motivation>
