207  template < typename ContainerType >
208  struct to_list
209  {
210  // Creates and returns a Python list from the elements copied
211  // from a STL container. The ContainerType must be a container
212  // with STL iterators defined on it.
213  // It may contain any type of object supported by the
214  // boost::python::object constructor.
215  static boost::python::object makeobject (ContainerType const& c)
216  {
217  boost::python::list result;
218  typename ContainerType::const_iterator i = c.begin();
219  typename ContainerType::const_iterator iEnd = c.end();
220  for( ; i != iEnd; ++i) {
221  result.append(*i);
222  }
223  return result;
224  }
225  static PyObject* convert (ContainerType const& c)
226  {
227  return boost::python::incref(makeobject(c).ptr());
228  }
229  };

400  // Appears to return @a obj_ptr if it is type of Python sequence
401  // that can be convertible to C++ container.
402  static void* convertible(PyObject* obj_ptr)
403  {
404  using namespace boost::python;
405  handle<> py_hdl(obj_ptr);
406  if (PyErr_Occurred()) {
407  PyErr_Clear();
408  return 0;
409  }
410  object py_obj(py_hdl);
411  incref(obj_ptr); // incr refcount, because ~object decrements it
412  // Accept single values.
413  if (PyBool_Check(obj_ptr)
414  || PyInt_Check(obj_ptr)
415  || PyLong_Check(obj_ptr)
416  || PyFloat_Check(obj_ptr)
417  || PyComplex_Check(obj_ptr)
418  || PyString_Check(obj_ptr)) {
419  extract<container_element_type> elem_proxy(py_obj);
420  if (!elem_proxy.check()) return 0;
421  return obj_ptr;
422  }
423  // An array scalar is accepted.
424  if (PycArrayScalarCheck(obj_ptr)) {
425  return obj_ptr;
426  }
427  // Get the sequence object.
428  // It can be a numarray/numpy scalar in which case
429  // it fills py_obj with a flattened array.
430  if (! getSeqObject (py_obj)) {
431  return 0;
432  }
433  // Check the sequence.
434  // It must be convertible to an iterator.
435  handle<> obj_iter(allow_null(PyObject_GetIter(py_obj.ptr())));
436  if (!obj_iter.get()) {
437  PyErr_Clear();
438  return 0;
439  }
440  if (!check_convertibility (py_obj.ptr())) {
441  return 0;
442  }
443  return obj_ptr;
444  }

447  static void construct(
448  PyObject* obj_ptr,
449  boost::python::converter::rvalue_from_python_stage1_data* data)
450  {
451  using namespace boost::python;
452  using boost::python::converter::rvalue_from_python_storage;
453  void* storage = (
454  (rvalue_from_python_storage<ContainerType>*)
455  data)->storage.bytes;
456  new (storage) ContainerType();
457  data->convertible = storage;
458  ContainerType& result = *((ContainerType*)storage);
459  if (PyBool_Check(obj_ptr)
460  || PyInt_Check(obj_ptr)
461  || PyLong_Check(obj_ptr)
462  || PyFloat_Check(obj_ptr)
463  || PyComplex_Check(obj_ptr)
464  || PyString_Check(obj_ptr)
465  || PycArrayScalarCheck(obj_ptr)) {
466  extract<container_element_type> elem_proxy(obj_ptr);
467  ConversionPolicy::reserve(result, 1);
468  ConversionPolicy::set_value(result, 0, elem_proxy());
469  return;
470  }
471  handle<> py_hdl(obj_ptr);
472  object py_obj = object(py_hdl);
473  incref(obj_ptr); // incr refcount, because ~object decrements it
474  assert (getSeqObject (py_obj));
475  fill_container (result, py_obj.ptr());

477  // ConversionPolicy::set_value(result, 0,
478  // extract<container_element_type>(py_flat.attr("__getitem__")(0)));
479  }

481  // Constructs a C++ container from a Python sequence.
482  static ContainerType make_container(PyObject* obj_ptr)
483  {
484  ContainerType result;
485  fill_container (result, obj_ptr);
486  return result;

489  private:
490  static void fill_container (ContainerType& result, PyObject* obj_ptr)
491  {
492  using namespace boost::python;
493  int obj_size = PyObject_Length(obj_ptr);
494  handle<> obj_iter(PyObject_GetIter(obj_ptr));
495  ConversionPolicy::reserve(result, obj_size);
496  std::size_t i=0;
497  for(;;i++) {
498  handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
499  if (PyErr_Occurred()) throw_error_already_set();
500  if (!py_elem_hdl.get()) break; // end of iteration
501  object py_elem_obj(py_elem_hdl);
502  extract<container_element_type> elem_proxy(py_elem_obj);
503  ConversionPolicy::set_value(result, i, elem_proxy());
504  }
505  ConversionPolicy::assert_size(boost::type<ContainerType>(), i);
506  }

508  static bool check_convertibility(PyObject* obj_ptr)
509  {
510  using namespace boost::python;
511  handle<> obj_iter(allow_null(PyObject_GetIter(obj_ptr)));
512  if (!obj_iter.get()) { // must be convertible to an iterator
513  PyErr_Clear();
514  return false;
515  }
516  int obj_size = PyObject_Length(obj_ptr);
517  if (obj_size < 0) { // must be a measurable sequence
518  PyErr_Clear();
519  return false;
520  }
521  if (ConversionPolicy::check_convertibility_per_element()) {
522  if (!ConversionPolicy::check_size(
523  boost::type<ContainerType>(), obj_size)) return false;
524  // All elements in a range and array have the same type, so
525  // need to check the first element only.
526  bool is_same = PyRange_Check(obj_ptr) ||
527  (PySequence_Check(obj_ptr)
528  && !PyTuple_Check(obj_ptr) && !PyList_Check(obj_ptr));
529  int i = 0;
530  for (;;i++) {
531  handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
532  if (PyErr_Occurred()) {
533  PyErr_Clear();
534  return false;
535  }
536  if (!py_elem_hdl.get()) break; // end of iteration
537  object py_elem_obj(py_elem_hdl);
538  extract<container_element_type> elem_proxy(py_elem_obj);
539  if (!elem_proxy.check()) return false;
540  if (is_same) break; // all elements are of the same type
541  }
542  if (!is_same) assert(i == obj_size );
543  }
544  return true;
545  }
