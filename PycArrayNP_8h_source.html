101  template <typename T>
102  struct array_scalar_from_python

112  // Check if it is a type we can convert.
113  static void* convertible(PyObject* obj_ptr)
114  {
115  int type;
116  if (PycArrayScalarCheck(obj_ptr, type)) {
117  return obj_ptr;
118  }
119  return 0;
120  }
