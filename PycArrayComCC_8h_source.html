141  void ArrayCopy<T>::toPy (void* to, const T* from, uInt nr)
142  {
143  if (sizeof(T) == sizeof(typename TypeConvTraits<T>::python_type)) {
144  ::memcpy (to, from, nr*sizeof(T));
145  } else {
146  typename TypeConvTraits<T>::python_type* dst =
147  static_cast<typename TypeConvTraits<T>::python_type*>(to);
148  for (uInt i=0; i<nr; i++) {
149  dst[i] = from[i];
150  }
151  }
152  }

166  template <typename T>
167  Array<T> ArrayCopy<T>::toArray (const IPosition& shape,
168  void* data, bool copy)
169  {
170  // If the python array was contiguous, etc., we can directly use
171  // its data because the Array used is only temporary.
172  // However, if a copy of the Python array was made in PycArray.cc,
173  // we cannot do that because the Python copy is out of scope when
174  // the Array object gets used.
175  if (!copy) {
176  if (sizeof(T) == sizeof(typename TypeConvTraits<T>::python_type)) {
177  return Array<T> (shape, static_cast<T*>(data), SHARE);
178  }
179  }
180  Array<T> arr(shape);
181  fromPy (arr.data(), data, arr.size());
182  return arr;
183  }

268  ValueHolder makeArray (PyObject* obj_ptr, Bool copyData)
269  {
270  if (! PycArrayCheck(obj_ptr)) {
271  throw AipsError ("PycArray: python object is not an array");
272  }
273  PyArrayObject* po = (PyArrayObject*)obj_ptr;
274  boost::python::object obj;
275  bool docopy = copyData; // copy data if wanted or needed
276  if (! PyArray_ISCONTIGUOUS(po)
277  || ! PyArray_ISALIGNED(po)
278  || PyArray_ISBYTESWAPPED(po)) {
279  boost::python::handle<> py_hdl(obj_ptr);
280  boost::python::object py_obj(py_hdl);
281  // incr refcount, because ~object decrements it
282  boost::python::incref(obj_ptr);
283  obj = py_obj.attr("copy")();
284  po = (PyArrayObject*)(obj.ptr());
285  docopy = true;
286  }
287  // Swap axes, because Casacore has row minor and Python row major order.
288  // A scalar is treated as a vector with length 1.
289  int nd = po->nd;
290  IPosition shp(1, 1);
291  if (nd > 0) {
292  shp.resize (nd);
293  for (int i=0; i<nd; i++) {
294  shp[i] = po->dimensions[nd-i-1];
295  }
296  }
297  // Assert array is contiguous now.
298  // If the array is empty, numarray still sees it as non-contiguous.
299  if (shp.product() > 0) {
300  AlwaysAssert (PyArray_ISCONTIGUOUS(po)
301  && PyArray_ISALIGNED(po)
302  && !PyArray_ISBYTESWAPPED(po), AipsError);
303  }
304  // Create the correct array.
305  switch (po->descr->type_num) {
306  case NPY_BOOL:
307  return ValueHolder (ArrayCopy<Bool>::toArray(shp, po->data, docopy));
308  case NPY_INT16:
309  return ValueHolder (ArrayCopy<Short>::toArray(shp, po->data, docopy));
310  case NPY_UINT16:
311  return ValueHolder (ArrayCopy<uShort>::toArray(shp, po->data, docopy));
312  case NPY_INT32:
313  return ValueHolder (ArrayCopy<Int>::toArray(shp, po->data, docopy));
314  case NPY_UINT32:
315  return ValueHolder (ArrayCopy<uInt>::toArray(shp, po->data, docopy));
316  case NPY_INT64:
317  return ValueHolder (ArrayCopy<Int64>::toArray(shp, po->data, docopy));
318  case NPY_FLOAT32:
319  return ValueHolder (ArrayCopy<Float>::toArray(shp, po->data, docopy));
320  case NPY_FLOAT64:
321  return ValueHolder (ArrayCopy<Double>::toArray(shp, po->data, docopy));
322  case NPY_COMPLEX64:
323  return ValueHolder (ArrayCopy<Complex>::toArray(shp, po->data, docopy));
324  case NPY_COMPLEX128:
325  return ValueHolder (ArrayCopy<DComplex>::toArray(shp, po->data, docopy));
326  case NPY_OBJECT:
327  return ValueHolder (ArrayCopy<String>::toArray(shp, po->data, docopy));
328  default:
329  // Some types can be the same as other types, so they cannot
330  // be used in the switch (compiler complains).
331  // This is true for BYTE and SBYTE which can equal to BOOL in numarray.
332  // Similarly for STRING which exists for numpy and is set to
333  // INT for numarray.
334  if (po->descr->type_num == NPY_UINT64) {
335  Array<uInt64> arr = ArrayCopy<uInt64>::toArray(shp, po->data, False);
336  Array<Int64> res(arr.shape());
337  convertArray (res, arr);
338  return ValueHolder(res);
339  } else if (po->descr->type_num == NPY_INT8) {
340  Array<Char> arr = ArrayCopy<Char>::toArray(shp, po->data, False);
341  Array<Short> res(arr.shape());
342  convertArray (res, arr);
343  return ValueHolder(res);
344  } else if (po->descr->type_num == NPY_UINT8) {
345  // Copy using Char, because uChar is mapped to Short in the Traits.
346  Array<Char> arr = ArrayCopy<Char>::toArray(shp, po->data, False);
347  Array<Short> res(arr.shape());
348  void* varr = &arr;
349  Array<uChar>* uarr = static_cast<Array<uChar>*>(varr);
350  convertArray (res, *uarr);
351  return ValueHolder(res);
352  } else if (po->descr->type_num == NPY_STRING) {
353  int slen = 0;
354  if (nd > 0) {
355  slen = po->strides[nd-1];
356  }
357  return ValueHolder (ArrayCopyStr_toArray(shp, po->data, slen));
358  }
359  break;
360  }
361  throw AipsError ("PycArray: unknown python array data type");
362  }
