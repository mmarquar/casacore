44 // <summary>
45 // A global enum used by some Array constructors.
46 // </summary>
47 // <synopsis>
48 // StorageInitPolicy is used in functions where an array is formed from
49 // a shape and an ordinary pointer. This enum should be in Array but that
50 // causes gcc to be unhappy.
51 // </synopsis>
52 enum StorageInitPolicy {
53  // COPY is used when an internal copy of the storage is to be made.
54  // The array is NOT responsible for deleting the external storage.
55  COPY,
56  // TAKE_OVER is used to indicate that the Array should just use the
57  // external storage (i.e., no copy is made). The Array class is now
58  // responsible for deleting the storage (hence it must have come from
59  // a call to new[]).
60  TAKE_OVER,
61  // Share means that the Array will just use the pointer (no copy), however
62  // the Array will NOT delete it upon destruction.
63  SHARE};
