40 // Class to calculate statistics using the so-called Chauvenet criterion. This method
41 // iteratively calculates statistics by discarding outliers on the basis of Chauvenet's
42 // criterion, until the specified maximum number of iterations is reached, or the final
43 // iteration results in no additional points being discarded.
44 // Alternatively, one can specify a z score which indicates the number of standard deviations
45 // beyond which to discard points. In this case, no iterating is done.
46 
47 template <class AccumType, class InputIterator, class MaskIterator=const Bool*> class ChauvenetCriterionStatistics
48  : public ConstrainedRangeStatistics<AccumType, InputIterator, MaskIterator> {
49 public:
50 
51  // If <src>zscore</src> is not negative, use that value to discard outliers beyond
52  // zscore standard deviations from the mean, and compute statistics based on the
53  // remaining data. If <src>zscore</src> is negative, use Chauvenet's Criterion to
54  // determine which outliers to discard. <src>maxIterations</src> is the maximum
55  // number of iterations to use before stopping. If negative, continue iterating until the
56  // set zscore or Chauvenet's criterion is met (ie that there are no remaining outliers).
57  ChauvenetCriterionStatistics(Double zscore=-1, Int maxIterations=0);
58 
59  virtual ~ChauvenetCriterionStatistics();

66  // get the algorithm that this object uses for computing stats
