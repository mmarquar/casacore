\usepackage{longtable}
\usepackage{array}
\usepackage{html}
\pagestyle{plain}

% The following marcos are necessary for the recipe system

% Define a macro for the overall recipe. We'd like the argument
% to go into a database for searching
\newenvironment{ahrecipe}[1]{
   \begin{latexonly}
   \begin{description}
   \setlength{\labelwidth}{1.0in} \setlength{\itemindent}{0.0in}
   \item[Recipe]{\bfseries #1}
   \end{description}
   \end{latexonly}
%
   \begin{htmlonly}
   \begin{description}
   \item[Recipe]{\bfseries #1}
   \end{description}
   \end{htmlonly}
   }{}

% What AIPS++ facilities does the recipe use?
\newcommand{\aruses}[1]
  {
   \begin{latexonly}
   \begin{description}
   \setlength{\labelwidth}{1.0in} \setlength{\itemindent}{0.0in}
   \item[Uses] #1
   \end{description}
   \end{latexonly}
%
   \begin{htmlonly}
   \begin{description}
   \item[Uses] #1
   \end{description}
   \end{htmlonly}
  }

% What are the results of the recipe?
\newcommand{\arresults}[1]
  {
   \begin{latexonly}
   \begin{description}
   \setlength{\labelwidth}{1.0in} \setlength{\itemindent}{0.0in}
   \item[Results] #1
   \end{description}
   \end{latexonly}
%
   \begin{htmlonly}
   \begin{description}

% Macro for lines of input (first argument) and comments (second
% argument
\newcommand{\arline}[2]
  { \begin{tabular}{>{\ttfamily \bfseries}p{2.5in}}
    #1
    \end{tabular}  & \begin{tabular}{>{\normalfont}p{3.2in}}
                     #2 
                     \end{tabular}
   \\
   \\ }

% Macro for output from a command
\newcommand{\aroutput}[1]
  {\\
   {\raggedright \mdseries #1} & \\ 
   \\}

% Macro for the example (consisting of \arlines's and \aroutputs)
\newenvironment{arexample}
  { \setlongtables
   \setlength{\LTleft}{0pt}
   \setlength{\LTright}{0pt}
   \setlength{\parindent}{0pt}
   \begin{longtable}[c]{p{2.5in}@{\hspace{0.3in}}p{3.2in}}
   \hline \textsl{\bfseries Commands and Results} & \textsl{\bfseries Comments}\\
   \\
   }{\end{longtable}}
