\newlength{\labellen}

\newcommand{\subsectionC}[4]{
  {\bfseries
   \settowidth{\labellen}{\thesubsubsection #1 --}
   \hspace{\the\labellen}
   \begin{tabular}{rl}
          Package & #2\\
          Module & #3\\
          Object & #4\\
    \end{tabular}
  }
  \flushleft\vspace{\baselineskip}
}
