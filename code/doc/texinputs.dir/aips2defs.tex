\newcommand{\subsectionC}[4]{
  \refstepcounter{subsection}
  \addcontentsline{toc}{subsection}{#1 -- Constructor}
   %{\protect\numberline{\subsectionname~\thesubsection} #1}
  {\flushleft\large\bfseries
  \begin{tabular}{r@{}r@{}l}
  
  \thesubsection ~ Constructor #1 -- & Package &~#2\\
                 & Module &~#3\\
                 & Object &~#4\\
  \end{tabular}
  }
  \subsectionmark{#1}\flushleft\vspace{\baselineskip}
}
