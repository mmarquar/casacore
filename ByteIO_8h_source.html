45 // <synopsis>
46 // ByteIO is the abstract base class for all classes doing IO on
47 // byte streams. Examples of derived classes are
48 // <linkto class=RegularFileIO>RegularFileIO</linkto> and
49 // <linkto class=MemoryIO>MemoryIO</linkto>.
50 // <p>
51 // ByteIO contains two enumerations, which define the possible
52 // open and seek options on byte streams. These enumerations
53 // are used throughout the IO framework.
54 // </synopsis>
55 
56 // <motivation>
57 // Make polymorphic operations on byte streams possible.
58 // </motivation>
59

61 class ByteIO
62 {
63 public:
64  // Define the possible ByteIO open options.
65  enum OpenOption {
66  Old=1,
67  // read/write; file must exist.
68  Update,
69  // read/write; create file if not exist.
70  Append,
71  // read/write; create file if not exist.
72  New,
73  // read/write; file may not exist yet.
74  NewNoReplace,
75  // read/write; delete file at close.
76  Scratch,
77  // read/write; file must exist; delete at close.
78  Delete
79  };
80 
81  // Define the possible seek options.
82  enum SeekOption {
83  // Seek from beginning of file.
84  Begin=1,
85  // Seek from current position.
86  Current,
87  // Seek from the end of the file.
88  End
89  };

97  // Write <src>size</src> bytes to the byte stream.
98  virtual void write (Int64 size, const void* buf) = 0;
99 
100  // Read <src>size</src> bytes from the byte stream. Returns the number of
101  // bytes actually read, or a negative number if an error occurred. Will also
102  // throw an Exception (AipsError) if the requested number of bytes could
103  // not be read unless throwException is set to False.
104  virtual Int64 read (Int64 size, void* buf, Bool throwException=True) = 0;
105 
106  // Reopen the underlying IO stream for read/write access.
107  // Nothing will be done if the stream is writable already.
108  // Otherwise it will be reopened and an exception will be thrown
109  // if it is not possible to reopen it for read/write access.
110  // The default implementation in this base class throws a "not possible"
111  // exception if a reopen has to be done.
112  virtual void reopenRW();
113 
114  // This function sets the position on the given offset.
115  // The seek option defines from which file position the seek is done.
116  // -1 is returned if not seekable.
117  // <group>
118  Int64 seek (Int offset, ByteIO::SeekOption = ByteIO::Begin);
119  Int64 seek (Int64 offset, ByteIO::SeekOption = ByteIO::Begin);
120  // </group>

122  // Flush the data to the file.
123  // The default implementation does nothing.
124  virtual void flush();

126  // Fsync the file (i.e. force the data to be physically written).

130  // Resync the file (i.e. empty the current buffer).
